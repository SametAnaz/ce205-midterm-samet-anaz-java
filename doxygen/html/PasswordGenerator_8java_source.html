77 public static String generatePassword(int length, boolean includeUppercase,
78 boolean includeLowercase, boolean includeDigits,
79 boolean includeSpecial) {
80 // Return empty string for zero or negative length
81 if (length <= 0) {
82 return "";
83 }
84
85 // Build the character set based on includes
86 StringBuilder charSetBuilder = new StringBuilder();
87
88 if (includeUppercase) charSetBuilder.append(UPPERCASE_CHARS);
89
90 if (includeLowercase) charSetBuilder.append(LOWERCASE_CHARS);
91
92 if (includeDigits) charSetBuilder.append(DIGIT_CHARS);
93
94 if (includeSpecial) charSetBuilder.append(SPECIAL_CHARS);
95
96 // If no character set is selected, use lowercase as default
97 String charSet = charSetBuilder.length() > 0 ? charSetBuilder.toString() : LOWERCASE_CHARS;
98 // Generate the password
99 StringBuilder password = new StringBuilder(length);
100 Random random = new Random();
101
102 // Ensure at least one character from each selected char set
103 if (length >= 1 && includeUppercase && password.length() < length) {
104 password.append(UPPERCASE_CHARS.charAt(random.nextInt(UPPERCASE_CHARS.length())));
105 }
106
107 if (length >= 2 && includeLowercase && password.length() < length) {
108 password.append(LOWERCASE_CHARS.charAt(random.nextInt(LOWERCASE_CHARS.length())));
109 }
110
111 if (length >= 3 && includeDigits && password.length() < length) {
112 password.append(DIGIT_CHARS.charAt(random.nextInt(DIGIT_CHARS.length())));
113 }
114
115 if (length >= 4 && includeSpecial && password.length() < length) {
116 password.append(SPECIAL_CHARS.charAt(random.nextInt(SPECIAL_CHARS.length())));
117 }
118
119 // Fill the rest with random characters from the selected set
120 while (password.length() < length) {
121 int index = random.nextInt(charSet.length());
122 password.append(charSet.charAt(index));
123 }
124
125 // Shuffle the characters to avoid predictable patterns
126 char[] passChars = password.toString().toCharArray();
127
128 for (int i = passChars.length - 1; i > 0; i--) {
129 int j = random.nextInt(i + 1);
130 char temp = passChars[i];
131 passChars[i] = passChars[j];
132 passChars[j] = temp;
133 }
134
135 return new String(passChars);
136 }

169 private static int[] computeLPSArray(String pattern) {
170 int m = pattern.length();
171 int[] lps = new int[m];
172 int len = 0; // Length of previous longest prefix suffix
173 int i = 1;
174 lps[0] = 0; // lps[0] is always 0
175
176 while (i < m) {
177 if (pattern.charAt(i) == pattern.charAt(len)) {
178 len++;
179 lps[i] = len;
180 i++;
181 } else {
182 if (len != 0) {
183 len = lps[len - 1];
184 } else {
185 lps[i] = 0;
186 i++;
187 }
188 }
189 }

206 public static int kmpSearch(String text, String pattern) {
207 if (text == null || pattern == null || pattern.isEmpty()) {
208 return -1;
209 }
210
211 int n = text.length();
212 int m = pattern.length();
213
214 if (m > n) {
215 return -1;
216 }
217
218 int[] lps = computeLPSArray(pattern);
219 int i = 0; // Index for text
220 int j = 0; // Index for pattern
221
222 while (i < n) {
223 if (pattern.charAt(j) == text.charAt(i)) {
224 i++;
225 j++;
226 }
227
228 if (j == m) {
229 return i - j; // Pattern found at index i-j
230 } else if (i < n && pattern.charAt(j) != text.charAt(i)) {
231 if (j != 0) {
232 j = lps[j - 1];
233 } else {
234 i++;
235 }
236 }
237 }
238
239 return -1; // Pattern not found

260 public static boolean isWeakPattern(String password) {
261 if (password == null || password.isEmpty()) {
262 return true;
263 }
264
265 // Common weak patterns
266 String[] weakPatterns = {
267 "123", "1234", "12345", "123456",
268 "abc", "abcd", "abcde",
269 "password", "pass", "pwd",
270 "qwerty", "admin", "user",
271 "000", "111", "aaa",
272 "password123", "admin123"
273 };
274 // Check case-insensitive
275 String lowerPassword = password.toLowerCase();
276
277 for (String pattern : weakPatterns) {
278 if (containsPattern(lowerPassword, pattern)) {
279 return true;
280 }
281 }
282
283 // Check for sequential numbers (ascending)
284 for (int i = 0; i < password.length() - 2; i++) {
285 if (Character.isDigit(password.charAt(i)) &&
286 Character.isDigit(password.charAt(i + 1)) &&
287 Character.isDigit(password.charAt(i + 2))) {
288 int d1 = password.charAt(i) - '0';
289 int d2 = password.charAt(i + 1) - '0';
290 int d3 = password.charAt(i + 2) - '0';
291
292 if (d2 == d1 + 1 && d3 == d2 + 1) {
293 return true; // Sequential like 123, 456, 789
294 }
295 }
296 }
297
298 // Check for repeated characters (3 or more)
299 for (int i = 0; i < password.length() - 2; i++) {
300 if (password.charAt(i) == password.charAt(i + 1) &&
301 password.charAt(i) == password.charAt(i + 2)) {
302 return true; // Repeated like aaa, 111
303 }
304 }

306 return false;

381 public static HuffmanNode buildHuffmanTree(Map<Character, Integer> frequencies) {
382 if (frequencies == null || frequencies.isEmpty()) {
383 return null;
384 }
385
386 PriorityQueue<HuffmanNode> pq = new PriorityQueue<>();
387
388 // Create leaf nodes for each character
389 for (Map.Entry<Character, Integer> entry : frequencies.entrySet()) {
390 pq.offer(new HuffmanNode(entry.getKey(), entry.getValue()));
391 }
392
393 // Build tree by combining nodes
394 while (pq.size() > 1) {
395 HuffmanNode left = pq.poll();
396 HuffmanNode right = pq.poll();
397 HuffmanNode parent = new HuffmanNode(
398 left.frequency + right.frequency, left, right
399 );
400 pq.offer(parent);
401 }
402
403 return pq.poll();
404 }
