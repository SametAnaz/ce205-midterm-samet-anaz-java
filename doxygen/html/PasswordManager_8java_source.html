627 public V put(K key, V value) {
628 if (key == null) {

630 }
631
632 // Check if resize is needed

637 int index = hash(key);
638 Entry<K, V> entry = buckets[index];
639
640 // Check if key already exists
641 while (entry != null) {
642 if (entry.key.equals(key)) {
643 V oldValue = entry.value;
644 entry.value = value;
645 return oldValue;
646 }
647
648 entry = entry.next;
649 }
650
651 // Add new entry at the beginning of the chain
652 Entry<K, V> newEntry = new Entry<>(key, value);
653 newEntry.next = buckets[index];
654
655 // Track collision
656 if (buckets[index] != null) {
657 collisionCount++;
658 }
659
660 buckets[index] = newEntry;
661 size++;
662 return null;
663 }

672 if (key == null) {
673 return null;
674 }
675

677 Entry<K, V> entry = buckets[index];
678
679 while (entry != null) {
680 if (entry.key.equals(key)) {
681 return entry.value;
682 }
683
684 entry = entry.next;
685 }

688 }

696 public V remove(K key) {
697 if (key == null) {
698 return null;
699 }
700
701 int index = hash(key);
702 Entry<K, V> entry = buckets[index];
703 Entry<K, V> prev = null;
704
705 while (entry != null) {
706 if (entry.key.equals(key)) {
707 if (prev == null) {
708 buckets[index] = entry.next;
709 } else {
710 prev.next = entry.next;
711 }
712
713 size--;
714 return entry.value;
715 }
716
717 prev = entry;
718 entry = entry.next;
719 }

721 return null;

881 private void heapSort(ServiceUsage[] arr) {
882 int n = arr.length;
883
884 // Build max heap
885 for (int i = n / 2 - 1; i >= 0; i--) {
886 heapify(arr, n, i);
887 }
888
889 // Extract elements from heap one by one
890 for (int i = n - 1; i > 0; i--) {
891 // Move current root to end
892 ServiceUsage temp = arr[0];
893 arr[0] = arr[i];
894 arr[i] = temp;
895 // Heapify the reduced heap
896 heapify(arr, i, 0);
897 }
898 }

907 private void heapify(ServiceUsage[] arr, int n, int i) {
908 int largest = i;
909 int left = 2 * i + 1;
910 int right = 2 * i + 2;
911
912 // Check if left child is larger than root
913 if (left < n && arr[left].compareTo(arr[largest]) > 0) {
914 largest = left;
915 }
916
917 // Check if right child is larger than current largest
918 if (right < n && arr[right].compareTo(arr[largest]) > 0) {
919 largest = right;
920 }
921
922 // If largest is not root
923 if (largest != i) {
924 ServiceUsage swap = arr[i];
925 arr[i] = arr[largest];
926 arr[largest] = swap;
927 // Recursively heapify the affected subtree
928 heapify(arr, n, largest);
929 }
930 }
